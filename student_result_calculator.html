import csv
import os

class StudentResultCalculator:
    def __init__(self, filepath='students_records.csv'):
        self.filepath = filepath
        # Validate existence of persistent file structure
        if not os.path.exists(self.filepath):
            self.create_default_file()

    def calculate_grades(self, math, sci, eng, hist, cs):
        """Evaluate averages and standard letter conversions"""
        total = math + sci + eng + hist + cs
        avg = total / 5.0
        
        if avg >= 90: grade = 'A'
        elif avg >= 80: grade = 'B'
        elif avg >= 70: grade = 'C'
        elif avg >= 60: grade = 'D'
        else: grade = 'F'
        
        status = 'Pass' if grade != 'F' else 'Fail'
        return round(avg, 2), grade, status

    def append_record(self, name, roll_id, grades):
        """Write calculation records securely to CSV storage"""
        avg, grade, status = self.calculate_grades(*grades)
        
        with open(self.filepath, mode='a', newline='', encoding='utf-8') as file:
            writer = csv.writer(file)
            writer.writerow([name, roll_id, *grades, avg, grade, status])
        return True
